### Loop

```rust
loop {
    println!("Loop forever!");
}
```

### While

```rust
let mut x = 5; // mut x: i32
let mut done = false; // mut done: bool
while !done {
    x += x - 3;
    println!("{}", x);
    if x % 5 == 0 {
        done = true;
    }
}
```

### For

```rust
for x in 0..10 {
    println!("{}", x); // x: i32
}
```

```rust
for (i,j) in (5..10).enumerate() {
    println!("i = {} and j = {}", i, j);
}
```

```rust
let lines = "line 0\nline 1".lines();
for (linenumber, line) in lines.enumerate() {
    println!("{}: {}", linenumber, line);
}
```

### Ending iteration early

```rust
let mut x = 5;
let mut done = false;
while !done {
    x += x - 3;
    println!("{}", x);
    if x % 5 == 0 {
        done = true;
    }
}
```

```rust
let mut x = 5;
loop {
    x += x - 3;
    println!("{}", x);
    if x % 5 == 0 { break; }
}
```

```rust
for x in 0..10 {
    if x % 2 == 0 { continue; }
    println!("{}", x);
}
```

### Loop labels

```rust
'outer: for x in 0..10 {
    'inner: for y in 0..10 {
        if x % 2 == 0 { continue 'outer; } // continues the loop over x
        if y % 2 == 0 { continue 'inner; } // continues the loop over y
        println!("x: {}, y: {}", x, y);
    }
}
```
